fn extract_keys(json: &Value) {
    match json {
        Value::Object(map) => {
            for (key, value) in map.iter() {
                println!("{}", key);
                extract_keys(value)
            }
        }
        Value::Array(vec) => {
            for value in vec.iter() {
               extract_keys(value);
            }
        }
        _ => (),
    }
}
